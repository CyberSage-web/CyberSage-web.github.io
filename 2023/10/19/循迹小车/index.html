二、工作原理

1、PWM调速: 脉冲宽度调制，PWM波形如下所示。

2、占空比：指高电平在一个周期中所占比例。对于单片机小车直流电机的调速，可以通过定时器设置定时初值，在中断服务函数中对占空比进行调整，从而实现电机的转速控制。

3、红外循迹：模块正式连接后，传感器开始工作。传感器有两个检测灯，一个灯用于发射光，另一个灯用于检测反射光的强度来判断是否检测到黑线。如果检测到黑线，黑色对光线的反射较弱，红外光无法反射回来，则模块指示灯熄灭，输出高电平（1）。未检测到黑线时，传感器发射出的红外光大部分被反射回来，模块感应到传感器反射回来的红外光，指示灯亮，输出低电平（0）。

四、代码部署

#include <reg51.h>					//通用51头文件

sbit ENA	= P1^0;
sbit IN1	=	P1^3;
sbit IN2	= P1^4;
sbit IN3	=	P1^5;
sbit IN4 	= P1^6;
sbit ENB	= P1^1;						//L298N引脚声明


sbit LEFT_IR		= P3^6;			//左内红外  D3 P36
sbit RIGHT_IR		= P2^0;			//右内红外	D2 P20
sbit LEFT_IR2		= P3^7;			//左外红外	D4 P37
sbit RIGHT_IR2	= P2^1;			//右外红外  D1 P21

unsigned char PWMA,PWMB;		//左电机占空比，有电机占空比
void xunji();								//循迹函数
void turnleft();						//左转函数
void turnleft2();						//左转函数2：大角度转弯
void turnright();						//右转函数
void turnright2();					//右转函数2：大角度转弯
void straight();						//直行函数
void stop();								//停车函数
void delayms(unsigned int);

void Timer0Init(void)		
{
    TMOD |= 0x01;							//设置定时器模式
    TL0 = (65536-100)%256;		//设置定时初始值
    TH0 = (65536-100)/256;		//设置定时初始值
    TR0 = 1;									//定时器0开始计时
    ET0 = 1;									//定时器0中断使能
    EA = 1;										//中断使能
}
static unsigned char jiaodu;						//转向标志
unsigned char end;
void main()
{
    Timer0Init();							//定时器0初始化
    while(1)
    {	
        xunji();								//循环执行循迹函数
    }
}

void T0isp() interrupt 1		//定时器0中断服务函数，生成2路可调PWM
                                                        //占空比由PWMA和PWMB（0-100）调节
{
    unsigned char i,j;
    TL0 = (65536-100)%256;		//设置定时初始值
    TH0 = (65536-100)/256;		//设置定时初始值
    i++;											//变量i用于控制PWMA占空比
    j++;											//变量j用于控制PWMB占空比
    if(i < PWMA)							//PWMA值以下，输出高电平
    {
        ENA = 1;								
    }
    else 											//PWMA-100范围，输出低电平
    {
        ENA = 0;
        if(i >= 100)
        {
            i = 0;
        }
    }
    if(j < PWMB)							//PWMB值以下，输出高电平
    {
        ENB = 1;
    }
    else 											//PWMB-100范围，输出低电平
    {
        ENB = 0;
        if(j >= 100)
        {
            j = 0;
        }
    }
}

void xunji()
{
    if(LEFT_IR == 0 && RIGHT_IR == 0	&& LEFT_IR2 == 0 && RIGHT_IR2 == 0&&end <= 1) //4个探头均未压线
    {		
        switch(jiaodu)
        {
            case 0:straight();break;
            case 1:turnleft2();break;
            case 2:turnright2();break;
            default:stop();break;
        }	
    }
    if(LEFT_IR == 1 && RIGHT_IR == 0 && LEFT_IR2 == 0 && RIGHT_IR2 == 0 &&end <= 1) //左内探头压线，车子轻微右倾，缓慢左转
    {
        turnleft();										//缓慢左转
        jiaodu = 0;
    }
    if(LEFT_IR == 0 && RIGHT_IR == 0 && LEFT_IR2 == 1 && RIGHT_IR2 == 0&&end <= 1) //左外探头压线，车子右倾，大角度左转
    {
        turnleft2();										//大角度左转
        jiaodu = 1;
    }
    if(LEFT_IR == 1 && RIGHT_IR == 0 && LEFT_IR2 == 1 && RIGHT_IR2 == 0&&end <= 1) //左内外探头压线，左90度弯，大角度左转
    {
        turnleft2();										//大角度左转
        jiaodu = 1;
    }
        if(LEFT_IR == 1 && RIGHT_IR == 1 && LEFT_IR2 == 1 && RIGHT_IR2 == 0&&end <= 1) //左内外、右内探头压线，左90度弯，大角度左转
    {
        turnleft2();										//大角度左转
        jiaodu = 1;
    }
    if(LEFT_IR == 0 && RIGHT_IR == 1 && LEFT_IR2 == 0 && RIGHT_IR2 == 0&&end <= 1) //右内探头压线，车子轻微左倾，缓慢右转
    {
        turnright();										//缓慢右转
        jiaodu = 0;
    }
    if(LEFT_IR == 0 && RIGHT_IR == 0 && LEFT_IR2 == 0 && RIGHT_IR2 == 1&&end <= 1) //右外探头压线，车子左倾，大角度右转
    {
        turnright2();									//大角度右转
        jiaodu = 2;
    }
    if(LEFT_IR == 0 && RIGHT_IR == 1 && LEFT_IR2 == 0 && RIGHT_IR2 == 1&&end <= 1) //右内外探头压线，左90度弯，大角度左转
    {
        turnleft2();									//大角度右转
        jiaodu = 2;
    }
    if(LEFT_IR == 1 && RIGHT_IR == 1 && LEFT_IR2 == 0 && RIGHT_IR2 == 1&&end <= 1) //右内外、左内探头压线，左90度弯，大角度左转
    {
        turnleft2();									//大角度右转
        jiaodu = 2;
    }
    if(LEFT_IR == 1 && RIGHT_IR == 1 && LEFT_IR2 == 1 && RIGHT_IR2 == 1) //4个探头均压线,停车
    {	
        stop();
    jiaodu = 3;
        end ++;
        if(end == 1)
        {
            turnleft2();
            delayms(500);    
            while(LEFT_IR == 0) turnleft2();
        }
        if(end > 1) end = 2;
    }
}

void straight()											//直行，PWM 60
{
    IN1 = 1;
    IN2 = 0;
    PWMA = 60;												//左轮正转，PWM=60
    IN3 = 1;
    IN4 = 0;
    PWMB = 60;												//右轮正转，PWM=60
}
void turnleft()											//左转，PWM=25
{
    IN1 = 0;
    IN2 = 0;
    PWMA = 0;													//左轮停转
    IN3 = 1;
    IN4 = 0;
    PWMB = 60;												//右轮正转，PWM=25
}
void turnleft2()										//左转:大角度转弯，PWM=25
{
    IN1 = 0;
    IN2 = 1;
    PWMA = 60;												//左轮反转，PWM=25
    IN3 = 1;
    IN4 = 0;
    PWMB = 60;												//右轮正转，PWM=25
}
void turnright()										//右转，PWM=25
{
    IN1 	= 1;
    IN2 	= 0;
    PWMA 	= 60;												//左轮正转，PWM=25
    IN3 	= 0;
    IN4 	= 0;
    PWMB	 = 0;												//右轮停转
}
void turnright2()										//右转：大角度转弯，PWM=25
{
    IN1 = 1;
    IN2 = 0;
    PWMA = 60;												//左轮正转，PWM=25
    IN3 = 0;
    IN4 = 1;
    PWMB = 60;												//右轮反转，PWM=25
}
void stop()													//停车，PWM=0
{
    IN1 = 0;
    IN2 = 0;
    PWMA = 0;													//左轮停转
    IN3 = 0;
    IN4 = 0;
    PWMB = 0;													//右轮停转
}
void delayms(unsigned int ms)
{
    unsigned int i,j;
    for(i = 0;i < ms;i++)
        for(j = 0;j < 112;j++);
}
